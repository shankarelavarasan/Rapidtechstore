# GitHub Repository Setup Commands
# Run these commands after creating your GitHub repository

# 1. Add GitHub remote (replace YOUR_USERNAME with your GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/rapid-tech-store-investor-pitch.git

# 2. Set main branch name
git branch -M main

# 3. Push to GitHub
git push -u origin main

# Alternative: If you prefer SSH (after setting up SSH keys)
# git remote add origin git@github.com:YOUR_USERNAME/rapid-tech-store-investor-pitch.git
# git push -u origin main
